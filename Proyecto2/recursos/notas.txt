# comandos kubernetes

# crear cluster
gcloud container clusters create k8s-demo --num-nodes=2 --tags=allin,allout --machine-type=n1-standard-2 --no-enable-network-policy

# conectarme al cluster
gcloud container clusters get-credentials cluster-sopes1 --zone us-central1-c

# traer toda la info del cluster
kubectl get all
